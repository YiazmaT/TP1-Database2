        DefaultTableModel dtm = (DefaultTableModel) produtoAtomicoTable.getModel();
        String [] linha = new String[2];
        
        dtm.setRowCount(0);
        for(ProdutoAtomico atual : p){
            linha[0] = atual.getNome();
            linha[1] = atual.getUnidade();
            dtm.addRow(linha);
        }
    }
