    public void preencherTabela(){
        DefaultTableModel dtm = (DefaultTableModel) estoqueTable.getModel();
        ArrayList<ProdutoAtomico> p;
        String [] linha = new String[4];
        for(Loja lojaAtual: lojas){
            p = BancoDeDados.estoqueDeUmaLoja(lojaAtual.getId_lanchonete());
            if(!p.isEmpty()){
                for(ProdutoAtomico pas : p){
                    linha[0] = pas.getNome();
                    linha[1] = pas.getUnidade();
                    linha[2] = String.valueOf(pas.getQuantidade());
                    linha[3] = lojaAtual.getNome();
                    dtm.addRow(linha);
                }
            } 
        }
    }
