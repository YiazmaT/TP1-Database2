    private void jButton1ActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_jButton1ActionPerformed
        //pegar e validar nome;
        String nome = nomeField.getText();
        if(nome.equals("")) {this.erro(1); return;}
        if(nome.length() > 50) {this.erro(6); return;}
        //fazer pra todos!\/
        nome = nome.toLowerCase();
        if(!nome.matches("([a-zA-Z ]|â|à|ã|ä|á|Á|ê|ë|é|É|î|ï|í|Í|ô|õ|ö|ó|Ó|û|ü|ú|Ú|'|ç){1,50}")){this.erro(1); return;}
        nome = nome.toUpperCase();
        
        //pegar e validar telefone
        String telefone = telefoneField.getText();
        if(telefone.contains(" ")){this.erro(3); return;}
        
        BancoDeDados.updateFuncionario(nome, telefone, this.func);
        this.dispose();
    }//GEN-LAST:event_jButton1ActionPerformed

    private void erro(int e){
        switch(e){
            case 1: JOptionPane.showMessageDialog(null, "Nome Inválido!! Utilize apenas letras."); break;
            case 2: JOptionPane.showMessageDialog(null, "CPF Inválido!!"); break;
            case 3: JOptionPane.showMessageDialog(null, "Telefone Inválido!! Utilize apenas números."); break;
            case 4: JOptionPane.showMessageDialog(null, "Login Inválido!! Utilize apenas letras sem acento e números."); break;
            case 5: JOptionPane.showMessageDialog(null, "Este Login já esta cadastrado no Sistema!!"); break;
            case 6: JOptionPane.showMessageDialog(null, "Nome pode ter no máximo 50 caracteres!!"); break;
            case 7: JOptionPane.showMessageDialog(null, "Login pode ter no máximo 25 caracteres!!"); break;
            case 8: JOptionPane.showMessageDialog(null, "Senha pode ter no máximo 50 caracteres!!"); break;
            case 9: JOptionPane.showMessageDialog(null, "Este CPF já foi cadastrado no Sistema!!"); break;
        }
    }
    
    public void preencherCampos(){
        if(this.func.getTipo().equals("caixa")){
            Caixa caixa = BancoDeDados.informacoesDeUmCaixa(this.func.getId());
            nomeField.setText(caixa.getNome());
            telefoneField.setText(caixa.getTelefone());
        }
        if(this.func.getTipo().equals("gerente")){
            Gerente gerente = BancoDeDados.informacoesDeUmGerente(this.func.getId());
            nomeField.setText(gerente.getNome());
            telefoneField.setText(gerente.getTelefone());
        }
        if(this.func.getTipo().equals("faxineiro")){
            Faxineiro faxineiro = BancoDeDados.informacoesDeUmFaxineiro(this.func.getId());
            nomeField.setText(faxineiro.getNome());
            telefoneField.setText(faxineiro.getTelefone());
        }
        if(this.func.getTipo().equals("cozinheiro")){
            Cozinheiro cozinheiro = BancoDeDados.informacoesDeUmCozinheiro(this.func.getId());
            nomeField.setText(cozinheiro.getNome());
            telefoneField.setText(cozinheiro.getTelefone());
        }
    }
