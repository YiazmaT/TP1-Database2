    private void jButton1ActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_jButton1ActionPerformed
        //pegar e validar quantidade;
        String q = quantidadeField.getText();
        if(q.equals("")) {erro(1); return;}
        if(!q.matches("[0-9]{1,50}")){this.erro(1); return;}
        int quantidade = Integer.parseInt(q);
        if(quantidade <= 0) {this.erro(1); return;}
        
        //pegar e validar codigo produto;
        String c = codigoField.getText();
        if(c.equals("")) {erro(2); return;}
        if(!c.matches("[0-9]{1,50}")){this.erro(2); return;}
        int codigo = Integer.parseInt(c);
        if(codigo <= 0) {this.erro(2); return;}
        
        //veririfcar se o produto j치 est치 na nota;
        for(Produto atual : this.produtoComprados){
            if(atual.getId_produto() == codigo){
                this.erro(4);
                return;
            }
        }
        
        //verificar se o c칩digo existe
        Produto seraAdicionado = null;
        for(Produto atual : this.produtos){
            if(atual.getId_produto() == codigo){
                seraAdicionado = atual;
                seraAdicionado.setQuantidade(quantidade);
                break;
            }
        }
        if(seraAdicionado == null){
            this.erro(3);
            return;
        }
        
        boolean possuiEstoque = BancoDeDados.possuiEstoqueParaVender(seraAdicionado, pai.getIdUsuarioLoja());
        if(possuiEstoque == false){
            JOptionPane.showMessageDialog(null, "Estoque insuficiente!!");
            return;
        }
        
        //adicinoar na tabela;
        this.produtoComprados.add(seraAdicionado);
        listarTabelaComprados(this.produtoComprados);
    }//GEN-LAST:event_jButton1ActionPerformed

    private void jButton4ActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_jButton4ActionPerformed
        //pesquisar produtos que possuem a string como chave e addnar no array de pesquisa atual;
        String search = searchField.getText().toUpperCase();
        if(search.equals("")){
            produtosPesquisaAtual=produtos;
            this.listarTabelaProdutos(produtosPesquisaAtual);
            return;
        }
        else{
            produtosPesquisaAtual = new ArrayList<Produto>();
            for(Produto atual : produtos){
                if(atual.getNome().contains(search)){
                    produtosPesquisaAtual.add(atual);
                }
            }
            this.listarTabelaProdutos(produtosPesquisaAtual);
            return;
        }
    }//GEN-LAST:event_jButton4ActionPerformed

    private void jButton6ActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_jButton6ActionPerformed
        int linha = produtosNota.getSelectedRow();
        if(linha<0) return;
        
        String n = JOptionPane.showInputDialog("Introduza a nova quantidade:");
        if(n.equals("")) {erro(1); return;}
        if(!n.matches("[0-9]{1,50}")){this.erro(1); return;}
        int novaQuantidade = Integer.parseInt(n);
        if(novaQuantidade <=0) {this.erro(1); return;}
        
        int copia = this.produtoComprados.get(linha).getQuantidade();
        this.produtoComprados.get(linha).setQuantidade(novaQuantidade);
        boolean possuiEstoque = BancoDeDados.possuiEstoqueParaVender(this.produtoComprados.get(linha), pai.getIdUsuarioLoja());
        if(possuiEstoque == false){
            JOptionPane.showMessageDialog(null, "Estoque insuficiente");
            this.produtoComprados.get(linha).setQuantidade(copia);
            return;
        }
        
        this.listarTabelaComprados(this.produtoComprados);
        return;
    }//GEN-LAST:event_jButton6ActionPerformed

    public void erro(int e){
        switch(e){
            case 1: JOptionPane.showMessageDialog(null, "Quantidade Inv치lida!!"); break;
            case 2: JOptionPane.showMessageDialog(null, "C칩digo do produto Inv치lido!!"); break;
            case 3: JOptionPane.showMessageDialog(null, "C칩digo do produto Inexistente!!"); break;
            case 4: JOptionPane.showMessageDialog(null, "Produto j치 existente na nota!!"); break;
        }
    }
    
    public void listarTabelaProdutos(ArrayList<Produto> p){
        String[] s = new String[1];
        DefaultTableModel dtm = (DefaultTableModel) tableProdutos.getModel();
        dtm.setRowCount(0);
        for(Produto atual : p){
            s[0] = atual.getNome();
            dtm.addRow(s);
        }
    }
    
    public void listarTabelaComprados(ArrayList<Produto> p){
        DefaultTableModel dtm = (DefaultTableModel) produtosNota.getModel();
        dtm.setRowCount(0);
        String[] linha = new String [5];
        totalNota =0.0f;
        for(Produto atual : p){
            linha[0] = String.valueOf(atual.getId_produto());
            linha[1] = atual.getNome();
            linha[2] = String.valueOf(atual.getQuantidade());
            linha[3] = String.format("%.2f",atual.getValor());
            linha[4] = String.format("%.2f", atual.getValor()*atual.getQuantidade());
            totalNota += atual.getValor()*atual.getQuantidade();
            dtm.addRow(linha);
        }
        totalField.setText("R$"+String.format("%.2f", totalNota));
        if(p.isEmpty()){
            pagamentoButton.setEnabled(false);
            return;
        }
        pagamentoButton.setEnabled(true);
    }
